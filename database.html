// First Creating Recursive Realtionship in database SELECT category_id, category_name, parent_category_id FROM categories; WITH RECURSIVE category_tree (category_id, category_name, parent_category_id, depth, path) AS ( SELECT category_id, category_name, parent_category_id, 1, CAST(category_id AS VARCHAR(255)) FROM categories WHERE parent_category_id IS NULL UNION ALL SELECT c.category_id, c.category_name, c.parent_category_id, ct.depth + 1, CONCAT(ct.path, ' > ', c.category_id) FROM categories c JOIN category_tree ct ON c.parent_category_id = ct.category_id ) SELECT category_id, category_name, parent_category_id, depth, path FROM category_tree ORDER BY path;
